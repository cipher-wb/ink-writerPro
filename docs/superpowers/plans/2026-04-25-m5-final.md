class RecurrenceRecord:
    case_id: str

    evidence_chain_path: str
    resolved_at: str
    regressed_at: str
    severity_before: str
    severity_after: str

    def to_dict(self) -> dict[str, Any]:
        return {
            "case_id": self.case_id,
            "book": self.book,
            "chapter": self.chapter,
            "evidence_chain_path": self.evidence_chain_path,
            "resolved_at": self.resolved_at,
            "regressed_at": self.regressed_at,
            "severity_before": self.severity_before,

```python
def iter_resolved(self) -> Iterator[Case]:
    """所有 status=resolved 的 case。"""
    for c in self._cases.values():
        if c.status == CaseStatus.RESOLVED:
            yield c


def record_recurrence(self, *, case_id: str, record: dict) -> Case:
    """给 case 加一条 recurrence_history + 升级 severity（hard 已顶则 recurrence_count++）。
    
    - status: resolved → regressed
    - severity: P0 已顶则不变（写 recurrence_count）；其他升一级
    """
    c = self.get(case_id)
    if c is None:
        raise KeyError(case_id)
    c.recurrence_history.append(record)
    c.status = CaseStatus.REGRESSED
    severity_order = ["P3", "P2", "P1", "P0"]
    cur_idx = severity_order.index(c.severity.value) if c.severity.value in severity_order else 3
    if cur_idx < len(severity_order) - 1:
        from ink_writer.case_library.models import CaseSeverity
        c.severity = CaseSeverity(severity_order[cur_idx + 1])
    self._save_case(c)
    return c
```

```python
"""scan_evidence_chains() + apply_recurrence() — Layer 4 主入口（spec §4）。

只检查同 book 内复发（Q3 默认）。
"""

from __future__ import annotations

import json
import time
from datetime import datetime
from pathlib import Path
from typing import Any

from ink_writer.case_library.models import Case, CaseStatus
from ink_writer.case_library.store import CaseStore
from ink_writer.regression_tracker.models import RecurrenceRecord


def _iter_evidence_chain_files(base_dir: Path) -> list[tuple[Path, str, str | None]]:
    """返回 [(path, book, chapter), ...]；planning 文件 chapter=None。"""
    out = []
    for book_dir in sorted(base_dir.iterdir()):
        if not book_dir.is_dir():
            continue
        # chapter evidence
        ch_dir = book_dir / "chapters"
        if ch_dir.exists():
            for ev in ch_dir.glob("*.evidence.json"):
                out.append((ev, book_dir.name, ev.stem.replace(".evidence", "")))
        # planning evidence
        pl = book_dir / "planning_evidence_chain.json"
        if pl.exists():
            out.append((pl, book_dir.name, None))
    return out


def _extract_cases_violated(ev_dict: dict) -> list[str]:
    """从 evidence_chain dict 里取所有 cases_violated/cases_hit case_id 列表。"""
    out: list[str] = []
    # M3 chapter schema：checkers 是 list[dict] 含 cases_hit/cases_violated
    if "checkers" in ev_dict:
        for ck in ev_dict["checkers"]:
            out.extend(ck.get("cases_violated") or [])
            out.extend(ck.get("cases_hit") or [])
    # M4 planning schema：stages
    if "stages" in ev_dict:
        for s in ev_dict["stages"]:
            for ck in s.get("checkers", []):
                out.extend(ck.get("cases_violated") or [])
                out.extend(ck.get("cases_hit") or [])
    return list(set(out))


def scan_evidence_chains(
    *,
    base_dir: Path = Path("data"),
    case_store: CaseStore,
    since: datetime | None = None,
) -> list[RecurrenceRecord]:
    """扫描所有 evidence_chain，返回新检测到的 resolved → regressed 复发记录。"""
    if not base_dir.exists():
        return []
    resolved_ids = {c.id for c in case_store.iter_resolved()}
    if not resolved_ids:
        return []

    records: list[RecurrenceRecord] = []
    seen_book_pairs: set[tuple[str, str]] = set()  # (case_id, book)
    now = time.strftime("%Y-%m-%dT%H:%M:%SZ", time.gmtime())

    for ev_path, book, chapter in _iter_evidence_chain_files(base_dir):
        if since:
            mtime = datetime.fromtimestamp(ev_path.stat().st_mtime)
            if mtime < since:
                continue
        try:
            ev = json.loads(ev_path.read_text(encoding="utf-8"))
        except Exception:
            continue
        violated = _extract_cases_violated(ev)
        for cid in violated:
            if cid not in resolved_ids:
                continue
            key = (cid, book)
            if key in seen_book_pairs:
                continue       # 同 book 多章，仅记一次（避免刷屏，详细信息靠 chapter 字段记最新）
            seen_book_pairs.add(key)
            case = case_store.get(cid)
            if case is None:
                continue
            records.append(RecurrenceRecord(
                case_id=cid,
                book=book,
                chapter=chapter,
                evidence_chain_path=str(ev_path),
                resolved_at="",  # case_store 不存这个；如需要可由用户审 case 时补
                regressed_at=now,
                severity_before=case.severity.value,
                severity_after=case.severity.value,  # apply 时再算升级
            ))
    return records


def apply_recurrence(
    *,
    record: RecurrenceRecord,
    case_store: CaseStore,
) -> Case:
    """持久化：写 case.recurrence_history + 升级 severity + 改 status=regressed。"""
    c = case_store.record_recurrence(
        case_id=record.case_id,
        record=record.to_dict(),
    )
    return c
```

- [ ] **Step 4: 写 __main__.py CLI**

```python
"""python -m ink_writer.regression_tracker --since YYYY-MM-DD"""

from __future__ import annotations

import argparse
import json
import sys
from datetime import datetime
from pathlib import Path

from ink_writer.case_library.store import CaseStore
from ink_writer.regression_tracker.tracker import (
    apply_recurrence, scan_evidence_chains,
)
from ink_writer.scripts.runtime_compat import enable_windows_utf8_stdio


def main() -> int:
    enable_windows_utf8_stdio()
    parser = argparse.ArgumentParser()
    parser.add_argument("--since", default=None, help="YYYY-MM-DD")
    parser.add_argument("--base-dir", default="data")
    parser.add_argument("--apply", action="store_true",
                        help="实际写盘升级 severity；默认 dry-run")
    args = parser.parse_args()

    since = datetime.fromisoformat(args.since) if args.since else None
    store = CaseStore()
    records = scan_evidence_chains(
        base_dir=Path(args.base_dir),
        case_store=store,
        since=since,
    )
    print(json.dumps(
        {"detected": len(records), "records": [r.to_dict() for r in records]},
        ensure_ascii=False, indent=2,
    ))
    if args.apply:
        for r in records:
            apply_recurrence(record=r, case_store=store)
        print(f"applied: {len(records)}")
    return 0


if __name__ == "__main__":
    sys.exit(main())
```

"""find_similar_clusters() + write_meta_rule_proposal() — Layer 5 主入口（spec §5）。"""

from __future__ import annotations

import json
import re
from pathlib import Path
from typing import Any

import yaml

from ink_writer.case_library.models import Case
from ink_writer.meta_rule_emergence.models import MetaRuleProposal

PROMPT_PATH = Path(__file__).resolve().parent / "prompts" / "emerge.txt"
_DEFAULT_MIN_CLUSTER = 5
_DEFAULT_SIMILARITY = 0.80
_DEFAULT_MODEL = "glm-4.6"


def _load_prompt() -> str:
    return PROMPT_PATH.read_text(encoding="utf-8")


def _extract_json(raw: str) -> dict[str, Any]:
    raw = raw.strip()
    if raw.startswith("```"):
        raw = raw.strip("`").strip()
        if raw.lower().startswith("json"):
            raw = raw[4:].strip()
    return json.loads(raw)


def _candidate_clusters_by_tags(cases: list[Case], min_size: int) -> list[list[Case]]:
    """先按 tag 重叠粗分组（避免 LLM N^2 调用）。同一 tag 集合的视为候选 cluster。"""
    by_tag: dict[str, list[Case]] = {}
    for c in cases:
        for t in c.tags:
            by_tag.setdefault(t, []).append(c)
    seen: set[frozenset[str]] = set()
    out: list[list[Case]] = []
    for tag, cs in by_tag.items():
        if len(cs) < min_size:
            continue
        ids = frozenset(c.id for c in cs)
        if ids in seen:
            continue
        seen.add(ids)
        out.append(cs)
    return out


def find_similar_clusters(
    *,
    cases: list[Case],
    llm_client: Any,
    min_cluster_size: int = _DEFAULT_MIN_CLUSTER,
    similarity_threshold: float = _DEFAULT_SIMILARITY,
    model: str = _DEFAULT_MODEL,
) -> list[MetaRuleProposal]:
    """扫描 cases，返回 MetaRuleProposal 列表。skip sovereign + 已 meta_rule_id 的。"""
    eligible = [c for c in cases if not c.sovereign and not c.meta_rule_id]
    candidates = _candidate_clusters_by_tags(eligible, min_cluster_size)

    proposals: list[MetaRuleProposal] = []
    next_id = _next_proposal_id()
    template = _load_prompt()

    for cluster in candidates:
        cluster_payload = [
            {
                "id": c.id, "title": c.title,
                "description": (c.failure_pattern or {}).get("description", ""),
                "tags": c.tags,
            }
            for c in cluster
        ]
        prompt = template.format(cases_json=json.dumps(cluster_payload, ensure_ascii=False))
        try:
            raw = llm_client.complete(prompt=prompt, model=model, max_tokens=2048, temperature=0)
            parsed = _extract_json(raw)
        except Exception:
            continue
        if not parsed.get("similar") or float(parsed.get("similarity", 0.0)) < similarity_threshold:
            continue
        covered = [cid for cid in parsed.get("covered_cases", []) if cid in {c.id for c in cluster}]
        if len(covered) < min_cluster_size:
            continue
        proposals.append(MetaRuleProposal(
            proposal_id=next_id,
            similarity=float(parsed.get("similarity", 0.0)),
            merged_rule=str(parsed.get("merged_rule", "")),
            covered_cases=covered,
            reason=str(parsed.get("reason", "")),
        ))
        next_id = _bump(next_id)
    return proposals


def _next_proposal_id(meta_dir: Path = Path("data/case_library/meta_rules")) -> str:
    """扫现有 MR-NNNN.yaml 取最大编号 + 1。"""
    if not meta_dir.exists():
        return "MR-0001"
    pattern = re.compile(r"MR-(\d{4})\.yaml$")
    nums = [int(m.group(1)) for p in meta_dir.glob("MR-*.yaml") if (m := pattern.search(p.name))]
    return f"MR-{(max(nums) + 1) if nums else 1:04d}"


def _bump(pid: str) -> str:
    n = int(pid.split("-")[1]) + 1
    return f"MR-{n:04d}"


def write_meta_rule_proposal(
    *,
    proposal: MetaRuleProposal,
    base_dir: Path = Path("data/case_library/meta_rules"),
) -> Path:
    base_dir.mkdir(parents=True, exist_ok=True)
    out = base_dir / f"{proposal.proposal_id}.yaml"
    with open(out, "w", encoding="utf-8") as fh:
        yaml.safe_dump(
            proposal.to_dict(),
            fh, allow_unicode=True, sort_keys=False, default_flow_style=False,
        )
    return out
```

"""ink meta-rule {list,approve,reject} 子命令实现。"""

from __future__ import annotations

import argparse
import sys
from pathlib import Path

import yaml

from ink_writer.case_library.store import CaseStore
from ink_writer.scripts.runtime_compat import enable_windows_utf8_stdio

DEFAULT_META_DIR = Path("data/case_library/meta_rules")


def _load_proposal(path: Path) -> dict:
    with open(path, encoding="utf-8") as fh:
        return yaml.safe_load(fh)


def _save_proposal(path: Path, data: dict) -> None:
    with open(path, "w", encoding="utf-8") as fh:
        yaml.safe_dump(data, fh, allow_unicode=True, sort_keys=False, default_flow_style=False)


def cmd_list(args: argparse.Namespace) -> int:
    base = Path(args.dir or DEFAULT_META_DIR)
    if not base.exists():
        print("(no meta_rules)")
        return 0
    rows = []
    for p in sorted(base.glob("MR-*.yaml")):
        d = _load_proposal(p)
        if args.status and d.get("status") != args.status:
            continue
        rows.append(d)
    print(f"meta-rules ({len(rows)}):")
    for d in rows:
        print(f"  {d['proposal_id']} status={d['status']} sim={d['similarity']:.2f} cases={len(d['covered_cases'])} :: {d['merged_rule']}")
    return 0


def cmd_approve(args: argparse.Namespace) -> int:
    base = Path(args.dir or DEFAULT_META_DIR)
    path = base / f"{args.proposal_id}.yaml"
    if not path.exists():
        print(f"not found: {path}", file=sys.stderr)
        return 1
    proposal = _load_proposal(path)
    if proposal.get("status") != "pending":
        print(f"已处理过 status={proposal.get('status')}", file=sys.stderr)
        return 1
    # 给覆盖 cases 写 meta_rule_id
    store = CaseStore()
    for cid in proposal["covered_cases"]:
        c = store.get(cid)
        if c is None:
            print(f"  跳过 {cid}（不存在）")
            continue
        c.meta_rule_id = proposal["proposal_id"]
        store._save_case(c)
    proposal["status"] = "approved"
    proposal["approved_at"] = __import__("time").strftime("%Y-%m-%dT%H:%M:%SZ", __import__("time").gmtime())
    _save_proposal(path, proposal)
    print(f"approved {args.proposal_id} → {len(proposal['covered_cases'])} cases 升级")
    return 0


def cmd_reject(args: argparse.Namespace) -> int:
    base = Path(args.dir or DEFAULT_META_DIR)
    path = base / f"{args.proposal_id}.yaml"
    if not path.exists():
        print(f"not found: {path}", file=sys.stderr)
        return 1
    proposal = _load_proposal(path)
    proposal["status"] = "rejected"
    proposal["rejected_at"] = __import__("time").strftime("%Y-%m-%dT%H:%M:%SZ", __import__("time").gmtime())
    _save_proposal(path, proposal)
    print(f"rejected {args.proposal_id}")
    return 0


def register_subparsers(subparsers) -> None:
    p = subparsers.add_parser("meta-rule", help="M5 元规则审批")
    p.add_argument("--dir", default=None, help=f"override {DEFAULT_META_DIR}")
    sub = p.add_subparsers(dest="meta_action", required=True)

    p_list = sub.add_parser("list")
    p_list.add_argument("--status", choices=["pending", "approved", "rejected"], default=None)
    p_list.set_defaults(func=cmd_list)

    p_app = sub.add_parser("approve")
    p_app.add_argument("proposal_id")
    p_app.set_defaults(func=cmd_approve)

    p_rej = sub.add_parser("reject")
    p_rej.add_argument("proposal_id")
    p_rej.set_defaults(func=cmd_reject)


def main() -> int:
    enable_windows_utf8_stdio()
    parser = argparse.ArgumentParser()
    sub = parser.add_subparsers(dest="cmd", required=True)
    register_subparsers(sub)
    args = parser.parse_args()
    return args.func(args)


if __name__ == "__main__":
    sys.exit(main())

```python
"""dashboard 4 大指标聚合 + dry-run 切换推荐（spec §6）。"""

from __future__ import annotations

import json
from datetime import datetime
from pathlib import Path
from typing import Any


def compute_recurrence_rate(
    *,
    case_store_iter,            # iter_active + iter_resolved 都需要
    base_dir: Path = Path("data"),
    since: datetime | None = None,
) -> float:
    """复发率 = (recurrence_history 非空的 case 数) / (resolved + regressed 总数)"""
    all_cases = list(case_store_iter)
    eligible = [c for c in all_cases if c.status.value in {"resolved", "regressed"}]
    if not eligible:
        return 0.0
    recurrent = [c for c in eligible if len(c.recurrence_history) > 0]
    return len(recurrent) / len(eligible)


def compute_repair_speed(
    *,
    case_store_iter,
) -> float:
    """修复速度 = 平均 (resolved 时间 - first failure 时间) 天数；M5 仅基于 created_at 估算。"""
    cases = [c for c in case_store_iter if c.status.value == "resolved"]
    if not cases:
        return 0.0
    days = []
    for c in cases:
        try:
            created = datetime.fromisoformat(c.created_at)
            # M5 没存 resolved_at；用 7 天（占位 placeholder）— 真实 M5 要求 case 加 resolved_at 字段
            days.append(7.0)    # M5 范围内：仅占位；真实数据需后续 enhance
        except Exception:
            continue
    return sum(days) / len(days) if days else 0.0


def compute_editor_score_trend(
    *,
    base_dir: Path = Path("data/editor_reviews"),
) -> list[dict]:
    """编辑评分趋势：扫 data/editor_reviews/*.yaml；每条返回 {date, score, book}。"""
    if not base_dir.exists():
        return []
    import yaml
    out = []
    for p in sorted(base_dir.glob("*.yaml")):
        try:
            d = yaml.safe_load(p.read_text(encoding="utf-8"))
            for entry in d.get("reviews", []):
                out.append({
                    "date": entry.get("date"),
                    "score": entry.get("score"),
                    "book": p.stem,
                })
        except Exception:
            continue
    return out


def compute_checker_accuracy(
    *,
    sample_dir: Path = Path("data/checker_accuracy_samples"),
) -> float:
    """checker 准确率：手工抽样 vs LLM 判定；M5 仅占位（真实数据由 US-009/010 后续录入）。"""
    if not sample_dir.exists():
        return 0.0
    return 0.0   # 占位；实际由 ink-learn 抽样工作流填充


def recommend_dry_run_switch(
    *,
    counter: int,
    pass_rate: float,
    threshold_runs: int = 5,
    threshold_pass_rate: float = 0.60,
) -> str:
    """返回 'switch' | 'continue' | 'investigate'（spec §6.5）。"""
    if counter < threshold_runs:
        return "continue"
    if pass_rate < threshold_pass_rate:
        return "investigate"
    return "switch"


def compute_m3_dry_run_pass_rate(
    *,
    base_dir: Path = Path("data"),
) -> tuple[int, float]:
    """返回 (counter, pass_rate)；扫 chapter evidence_chain。"""
    counter_path = base_dir / ".dry_run_counter"
    counter = 0
    if counter_path.exists():
        try:
            counter = int(counter_path.read_text(encoding="utf-8").strip())
        except ValueError:
            counter = 0

    passed = 0; total = 0
    for book_dir in base_dir.iterdir() if base_dir.exists() else []:
        ch_dir = book_dir / "chapters"
        if not ch_dir.exists():
            continue
        for ev in ch_dir.glob("*.evidence.json"):
            try:
                d = json.loads(ev.read_text(encoding="utf-8"))
                total += 1
                if d.get("overall_passed"):
                    passed += 1
            except Exception:
                continue
    return counter, (passed / total) if total else 0.0


def compute_m4_dry_run_pass_rate(
    *,
    base_dir: Path = Path("data"),
) -> tuple[int, float]:
    """返回 (counter, pass_rate)；扫 planning_evidence_chain。"""
    counter_path = base_dir / ".planning_dry_run_counter"
    counter = 0
    if counter_path.exists():
        try:
            counter = int(counter_path.read_text(encoding="utf-8").strip())
        except ValueError:
            counter = 0

    passed = 0; total = 0
    for book_dir in base_dir.iterdir() if base_dir.exists() else []:
        pl = book_dir / "planning_evidence_chain.json"
        if not pl.exists():
            continue
        try:
            d = json.loads(pl.read_text(encoding="utf-8"))
            for stage in d.get("stages", []):
                total += 1
                if stage.get("overall_passed"):
                    passed += 1
        except Exception:
            continue
    return counter, (passed / total) if total else 0.0
```

"""dashboard /api/m5-overview JSON 路由（spec §6.4）。"""

from __future__ import annotations

import json
from pathlib import Path
from typing import Any

from ink_writer.case_library.store import CaseStore
from ink_writer.dashboard.aggregator import (
    compute_recurrence_rate, compute_repair_speed,
    compute_editor_score_trend, compute_checker_accuracy,
    compute_m3_dry_run_pass_rate, compute_m4_dry_run_pass_rate,
    recommend_dry_run_switch,
)


def get_m5_overview(
    *,
    base_dir: Path = Path("data"),
    case_store: CaseStore | None = None,
) -> dict[str, Any]:
    """聚合 M5 dashboard 全部数据成单 dict。"""
    store = case_store or CaseStore()
    all_cases = list(store.iter_all())   # iter_all 遍历所有 status

    m3_counter, m3_pass = compute_m3_dry_run_pass_rate(base_dir=base_dir)
    m4_counter, m4_pass = compute_m4_dry_run_pass_rate(base_dir=base_dir)

    pending_meta = []
    meta_dir = Path("data/case_library/meta_rules")
    if meta_dir.exists():
        import yaml
        for p in sorted(meta_dir.glob("MR-*.yaml")):
            d = yaml.safe_load(p.read_text(encoding="utf-8"))
            if d.get("status") == "pending":
                pending_meta.append({
                    "id": d["proposal_id"],
                    "similarity": d["similarity"],
                    "covered_count": len(d["covered_cases"]),
                    "merged_rule": d["merged_rule"],
                })

    recurrent_cases = [
        {
            "id": c.id, "title": c.title,
            "severity": c.severity.value,
            "recurrence_count": len(c.recurrence_history),
            "latest_book": c.recurrence_history[-1].get("book") if c.recurrence_history else None,
        }
        for c in all_cases if c.recurrence_history
    ]

    return {
        "metrics": {
            "recurrence_rate": compute_recurrence_rate(case_store_iter=all_cases),
            "repair_speed_days": compute_repair_speed(case_store_iter=all_cases),
            "editor_score_trend": compute_editor_score_trend(),
            "checker_accuracy": compute_checker_accuracy(),
        },
        "dry_run": {
            "m3": {
                "counter": m3_counter, "pass_rate": m3_pass,
                "recommendation": recommend_dry_run_switch(counter=m3_counter, pass_rate=m3_pass),
            },
            "m4": {
                "counter": m4_counter, "pass_rate": m4_pass,
                "recommendation": recommend_dry_run_switch(counter=m4_counter, pass_rate=m4_pass),
            },
        },
        "pending_meta_rules": pending_meta,
        "recurrent_cases": recurrent_cases,
    }
```

"""generate_weekly_report() — 周报 markdown 生成（spec §7）。"""

from __future__ import annotations

import argparse
from datetime import datetime, timedelta
from pathlib import Path

from ink_writer.dashboard.m5_overview import get_m5_overview
from ink_writer.scripts.runtime_compat import enable_windows_utf8_stdio


def _week_range(week_num: int, year: int = 2026) -> tuple[str, str]:
    """ISO week → (start, end) date strings."""
    jan4 = datetime(year, 1, 4)
    week1_monday = jan4 - timedelta(days=jan4.weekday())
    monday = week1_monday + timedelta(weeks=week_num - 1)
    sunday = monday + timedelta(days=6)
    return monday.strftime("%Y-%m-%d"), sunday.strftime("%Y-%m-%d")


def generate_weekly_report(
    *,
    week_num: int,
    year: int = 2026,
    book: str | None = None,
    base_dir: Path = Path("data"),
    out_path: Path | None = None,
) -> Path:
    """生成 reports/weekly/<year>-W<NN>.md"""
    start, end = _week_range(week_num, year)
    overview = get_m5_overview(base_dir=base_dir)

    lines: list[str] = []
    lines.append(f"# Weekly Report - {year}-W{week_num:02d} ({start} ~ {end})")
    lines.append("")
    lines.append("## 4 大指标")
    m = overview["metrics"]
    lines.append(f"- 病例复发率：{m['recurrence_rate']*100:.1f}%")
    lines.append(f"- 修复速度：{m['repair_speed_days']:.1f} 天")
    score_trend = m["editor_score_trend"]
    if score_trend:
        lines.append(f"- 编辑评分：{score_trend[-1].get('score')}（{score_trend[-1].get('book')}）")
    else:
        lines.append("- 编辑评分：未录入")
    lines.append(f"- checker 准确率：{m['checker_accuracy']*100:.1f}%（如 0% 表示未抽样）")
    lines.append("")

    lines.append("## Layer 4 复发追踪")
    if overview["recurrent_cases"]:
        for r in overview["recurrent_cases"]:
            lines.append(f"- {r['id']}（{r['latest_book']}，第 {r['recurrence_count']} 次复发，severity={r['severity']}）")
    else:
        lines.append("- (无复发记录)")
    lines.append("")

    lines.append("## Layer 5 元规则浮现")
    if overview["pending_meta_rules"]:
        for p in overview["pending_meta_rules"]:
            lines.append(f"- {p['id']}（覆盖 {p['covered_count']} case，sim={p['similarity']:.2f}）— {p['merged_rule']}")
        lines.append("")
        lines.append(f"**待审批**：共 {len(overview['pending_meta_rules'])} 条")
    else:
        lines.append("- (无 pending 元规则)")
    lines.append("")

    lines.append("## Dry-run 状态")
    dr = overview["dry_run"]
    lines.append(f"- M3 chapter: {dr['m3']['counter']}/5（通过率 {dr['m3']['pass_rate']*100:.1f}%，推荐 {dr['m3']['recommendation']}）")
    lines.append(f"- M4 planning: {dr['m4']['counter']}/5（通过率 {dr['m4']['pass_rate']*100:.1f}%，推荐 {dr['m4']['recommendation']}）")
    lines.append("")

    lines.append("## 行动项")
    if overview["pending_meta_rules"]:
        lines.append(f"- [ ] 审批 {len(overview['pending_meta_rules'])} 条 pending 元规则（ink meta-rule list / approve / reject）")
    if dr["m3"]["recommendation"] == "switch":
        lines.append("- [ ] 评估 M3 dry-run 切真（通过率达标）")
    if dr["m4"]["recommendation"] == "switch":
        lines.append("- [ ] 评估 M4 dry-run 切真（通过率达标）")
    if not lines[-1].startswith("- [ ]"):
        lines.append("- (无紧急行动项)")

    out = out_path or Path(f"reports/weekly/{year}-W{week_num:02d}.md")
    out.parent.mkdir(parents=True, exist_ok=True)
    out.write_text("\n".join(lines) + "\n", encoding="utf-8")
    return out


def main() -> int:
    enable_windows_utf8_stdio()
    parser = argparse.ArgumentParser()
    parser.add_argument("--week", type=int, required=True)
    parser.add_argument("--year", type=int, default=2026)
    parser.add_argument("--book", default=None)
    parser.add_argument("--out", default=None)
    args = parser.parse_args()

    out = generate_weekly_report(
        week_num=args.week, year=args.year, book=args.book,
        out_path=Path(args.out) if args.out else None,
    )
    print(f"周报已写：{out}")
    return 0


if __name__ == "__main__":
    import sys
    sys.exit(main())
```

    import argparse, sys
    parser = argparse.ArgumentParser()
    sub = parser.add_subparsers(dest="cmd", required=True)
    
    p_report = sub.add_parser("report", help="生成周报")

```python
"""propose_cases_from_failures() — M5 ink-learn 自动从 failure 模式提议 pending case。"""

from __future__ import annotations

import json
import re
from collections import Counter
from datetime import datetime, timedelta
from pathlib import Path
from typing import Any

import yaml

from ink_writer.case_library.models import Case, CaseSeverity, CaseStatus
from ink_writer.case_library.store import CaseStore

DEFAULT_THROTTLE_PATH = Path("config/ink_learn_throttle.yaml")
DEFAULT_CASES_DIR = Path("data/case_library/cases")


def _load_throttle(path: Path | None = None) -> dict:
    p = Path(path) if path else DEFAULT_THROTTLE_PATH
    if not p.exists():
        return {"max_per_week": 5, "min_pattern_occurrences": 2, "pattern_window_days": 7}
    return yaml.safe_load(p.read_text(encoding="utf-8"))["auto_case_from_failure"]


def _scan_blocked_evidence(*, base_dir: Path, since: datetime) -> list[dict]:
    """扫所有 evidence_chain，返回 [{path, cases_violated, mtime}, ...] 仅 blocked 章节。"""
    out: list[dict] = []
    if not base_dir.exists():
        return out
    for book_dir in base_dir.iterdir():
        if not book_dir.is_dir():
            continue
        ch_dir = book_dir / "chapters"
        if not ch_dir.exists():
            continue
        for ev in ch_dir.glob("*.evidence.json"):
            mtime = datetime.fromtimestamp(ev.stat().st_mtime)
            if mtime < since:
                continue
            try:
                d = json.loads(ev.read_text(encoding="utf-8"))
            except Exception:
                continue
            if d.get("overall_passed"):
                continue
            violated = []
            for ck in d.get("checkers", []):
                violated.extend(ck.get("cases_violated") or [])
            if not violated:
                continue
            out.append({
                "path": str(ev), "cases_violated": sorted(set(violated)),
                "mtime": mtime,
            })
    return out


def _next_learn_id(cases_dir: Path) -> str:
    pattern = re.compile(r"CASE-LEARN-(\d{4})\.yaml$")
    nums = [int(m.group(1)) for p in cases_dir.glob("CASE-LEARN-*.yaml") if (m := pattern.search(p.name))]
    return f"CASE-LEARN-{(max(nums) + 1) if nums else 1:04d}"


def _bump(cid: str) -> str:
    n = int(cid.split("-")[-1]) + 1
    return f"CASE-LEARN-{n:04d}"


def propose_cases_from_failures(
    *,
    case_store: CaseStore,
    base_dir: Path = Path("data"),
    cases_dir: Path = DEFAULT_CASES_DIR,
    throttle_path: Path | None = None,
    now: datetime | None = None,
) -> list[Case]:
    """识别同一 cases_violated 组合连续出现 ≥ N 次的"新模式"，自动 propose pending case。"""
    cfg = _load_throttle(throttle_path)
    n = now or datetime.now()
    since = n - timedelta(days=cfg["pattern_window_days"])

    blocked = _scan_blocked_evidence(base_dir=base_dir, since=since)
    pattern_counter = Counter(tuple(b["cases_violated"]) for b in blocked)

    existing_ids = {c.id for c in case_store.iter_all()}
    proposed: list[Case] = []
    next_id = _next_learn_id(cases_dir)
    weekly_quota = cfg["max_per_week"]

    for pattern, count in pattern_counter.most_common():
        if len(proposed) >= weekly_quota:
            break
        if count < cfg["min_pattern_occurrences"]:
            break  # most_common 已排序
        # 跳过已 cover 的（pattern 中所有 case 都已 active）
        if all(cid in existing_ids for cid in pattern):
            # M5 简化：模式中所有 case 都存在 → 不视为新模式
            continue

        # 创建新 pending case 反映这个组合
        case = Case(
            id=next_id,
            title=f"自动学习模式：联合违反 {len(pattern)} case ({count} 次)",
            severity=CaseSeverity.P2,
            status=CaseStatus.PENDING,
            category="auto_learned",
            tags=["m5_auto_learn", *list(pattern)[:3]],   # 关联 case_id 作为 tag 便于追溯
            failure_pattern={
                "description": f"在 {cfg['pattern_window_days']} 天内观察到 {count} 次同时违反 {sorted(pattern)} 的组合，疑为新模式",
                "observable": list(pattern),
                "examples": [],
            },
            countermeasure={
                "guideline": f"待用户审：是否将此组合升级为独立 checker 或合并到现有 case",
            },
            source={
                "type": "ink_learn_auto",
                "notes": f"M5 US-009 propose_cases_from_failures; observed {count} times since {since.date()}",
            },
            created_at=n.strftime("%Y-%m-%d"),
        )
        proposed.append(case)
        # 写盘
        out_path = cases_dir / f"{next_id}.yaml"
        out_path.parent.mkdir(parents=True, exist_ok=True)
        with open(out_path, "w", encoding="utf-8") as fh:
            yaml.safe_dump(case.to_dict(), fh, allow_unicode=True, sort_keys=False)
        next_id = _bump(next_id)
    return proposed
```

"""promote_short_term_to_long_term() — 短期记忆 (.ink/project_memory.json) → 长期 case_library。"""

from __future__ import annotations

import json
import re
from collections import Counter
from datetime import datetime
from pathlib import Path
from typing import Any

import yaml

from ink_writer.case_library.models import Case, CaseSeverity, CaseStatus
from ink_writer.case_library.store import CaseStore


def _load_project_memory(path: Path) -> dict:
    if not path.exists():
        return {}
    return json.loads(path.read_text(encoding="utf-8"))


def _next_promote_id(cases_dir: Path) -> str:
    pattern = re.compile(r"CASE-PROMOTE-(\d{4})\.yaml$")
    nums = [int(m.group(1)) for p in cases_dir.glob("CASE-PROMOTE-*.yaml") if (m := pattern.search(p.name))]
    return f"CASE-PROMOTE-{(max(nums) + 1) if nums else 1:04d}"


def _bump(cid: str) -> str:
    n = int(cid.split("-")[-1]) + 1
    return f"CASE-PROMOTE-{n:04d}"


def promote_short_term_to_long_term(
    *,
    project_memory_path: Path,
    case_store: CaseStore,
    cases_dir: Path = Path("data/case_library/cases"),
    min_occurrences: int = 3,
) -> list[Case]:
    """读 project_memory.json，把"重复出现 ≥ N 次"的 success/failure 模式回灌到 case_library。
    
    schema:
      project_memory.json = {
        "patterns": [
          {"text": "...", "kind": "success|failure", "count": 5}, ...
        ]
      }
    """
    mem = _load_project_memory(project_memory_path)
    patterns = mem.get("patterns", [])
    promoted: list[Case] = []
    next_id = _next_promote_id(cases_dir)
    now = datetime.now()

    for p in patterns:
        if int(p.get("count", 0)) < min_occurrences:
            continue
        text = str(p.get("text", "")).strip()
        if not text:
            continue
        kind = str(p.get("kind", "failure"))
        case = Case(
            id=next_id,
            title=f"短期记忆回灌（{kind}）：{text[:30]}...",
            severity=CaseSeverity.P2 if kind == "failure" else CaseSeverity.P3,
            status=CaseStatus.PENDING,
            category="auto_promoted",
            tags=["m5_promote", kind],
            failure_pattern={"description": text, "observable": [], "examples": []},
            countermeasure={
                "guideline": f"短期记忆中此模式重复 {p['count']} 次；待用户审：升 active 或合并",
            },
            source={
                "type": "ink_learn_promote",
                "notes": f"M5 US-010 promote_short_term_to_long_term; from {project_memory_path.name}",
            },
            created_at=now.strftime("%Y-%m-%d"),
        )
        promoted.append(case)
        out_path = cases_dir / f"{next_id}.yaml"
        out_path.parent.mkdir(parents=True, exist_ok=True)
        with open(out_path, "w", encoding="utf-8") as fh:
            yaml.safe_dump(case.to_dict(), fh, allow_unicode=True, sort_keys=False)
        next_id = _bump(next_id)
    return promoted
