```python
"""Loader for config/checker-thresholds.yaml.

Read once on ink-write startup (Q13 决议)；caller cache the dict.
"""
from __future__ import annotations

from functools import lru_cache
from pathlib import Path
from typing import Any

import yaml


class ThresholdsConfigError(Exception):
    """Raised when config/checker-thresholds.yaml is missing or malformed."""


def load_thresholds(path: Path | None = None) -> dict[str, Any]:
    """Load YAML from `config/checker-thresholds.yaml` (or override path)."""
    if path is None:
        path = Path("config/checker-thresholds.yaml")
    if not path.is_file():
        raise ThresholdsConfigError(f"thresholds config not found: {path}")
    try:
        with open(path, encoding="utf-8") as fp:
            data = yaml.safe_load(fp)
    except yaml.YAMLError as err:
        raise ThresholdsConfigError(f"failed to parse {path}: {err}") from err
    if not isinstance(data, dict):
        raise ThresholdsConfigError(f"top-level must be a mapping, got {type(data)}")
    return data
```

```python
"""EvidenceChain dataclass — per-chapter audit trail (spec §6.1)."""
from __future__ import annotations

from dataclasses import dataclass, field
from datetime import datetime, timezone
from typing import Any


@dataclass
class EvidenceChain:
    book: str
    chapter: str
    dry_run: bool = False
    outcome: str = "pending"  # pending | delivered | needs_human_review
    produced_at: str = field(
        default_factory=lambda: datetime.now(timezone.utc).isoformat()
    )

    # phase_evidence sub-records
    context_recalled_rules: int = 0
    context_recalled_chunks: int = 0
    context_recalled_cases: int = 0
    writer_prompt_hash: str = ""
    writer_model: str = "glm-4.6"
    writer_rounds: list[dict[str, Any]] = field(default_factory=list)
    checker_results: list[dict[str, Any]] = field(default_factory=list)
    polish_rounds: list[dict[str, Any]] = field(default_factory=list)

    # case_evidence_updates
    case_updates: list[dict[str, Any]] = field(default_factory=list)

    # human_overrides
    human_overrides: list[dict[str, Any]] = field(default_factory=list)

    def record_self_check(self, round_idx: int, compliance: dict[str, Any]) -> None:
        self.writer_rounds.append({
            "round": round_idx,
            "compliance_report": dict(compliance),
        })

    def record_checkers(self, round_idx: int, results: list[dict[str, Any]]) -> None:
        self.checker_results = list(results)  # 末轮覆盖前轮

    def record_polish(self, round_idx: int, case_id: str, result: str) -> None:
        self.polish_rounds.append({
            "round": round_idx,
            "case_id": case_id,
            "result": result,
        })

    def record_case_update(self, case_id: str, result: str, by: str) -> None:
        self.case_updates.append({
            "case_id": case_id,
            "result": result,
            "by": by,
        })

    def to_dict(self) -> dict[str, Any]:
        return {
            "$schema": "https://ink-writer/evidence_chain_v1",
            "book": self.book,
            "chapter": self.chapter,
            "produced_at": self.produced_at,
            "dry_run": self.dry_run,
            "outcome": self.outcome,
            "phase_evidence": {
                "context_agent": {
                    "recalled": {
                        "rules": self.context_recalled_rules,
                        "chunks": self.context_recalled_chunks,
                        "cases": self.context_recalled_cases,
                    },
                    "recall_quality_avg": None,  # M3 期 null
                },
                "writer_agent": {
                    "prompt_hash": self.writer_prompt_hash,
                    "model": self.writer_model,
                    "rounds": list(self.writer_rounds),
                },
                "checkers": list(self.checker_results),
                "polish_agent": {
                    "rewrite_rounds": len(self.polish_rounds),
                    "rewrite_drivers": list(self.polish_rounds),
                },
            },
            "case_evidence_updates": list(self.case_updates),
            "human_overrides": list(self.human_overrides),
        }
```

```python
"""Evidence chain writer + EvidenceChainMissingError (M3 强制必带 — Q5)."""
from __future__ import annotations

import json
from pathlib import Path

from ink_writer.evidence_chain.models import EvidenceChain


class EvidenceChainMissingError(Exception):
    """Raised when ink-write 章节交付时缺 evidence_chain.json."""


def _evidence_path(book: str, chapter: str, base_dir: Path | None = None) -> Path:
    if base_dir is None:
        base_dir = Path("data")
    return base_dir / book / "chapters" / f"{chapter}.evidence.json"


def write_evidence_chain(
    *,
    book: str,
    chapter: str,
    evidence: EvidenceChain,
    base_dir: Path | None = None,
) -> Path:
    out_path = _evidence_path(book, chapter, base_dir)
    out_path.parent.mkdir(parents=True, exist_ok=True)
    with open(out_path, "w", encoding="utf-8") as fp:
        json.dump(evidence.to_dict(), fp, ensure_ascii=False, indent=2)
    return out_path


def require_evidence_chain(
    *,
    book: str,
    chapter: str,
    base_dir: Path | None = None,
) -> Path:
    """Verify evidence_chain.json exists for given (book, chapter)；否则 raise。"""
    p = _evidence_path(book, chapter, base_dir)
    if not p.is_file():
        raise EvidenceChainMissingError(
            f"evidence_chain.json missing for {book}/{chapter}: expected {p}"
        )
    return p
```

```python
"""writer_self_check — chapter-level rule_compliance + cases evaluation (spec §3)."""
from __future__ import annotations

import json
import re
import statistics
from pathlib import Path
from typing import Any

from ink_writer.writer_self_check.models import ComplianceReport

_PROMPT_PATH = Path(__file__).parent / "prompts" / "self_check.txt"
_RULE_COMPLIANCE_THRESHOLD = 0.70  # spec §5.1 + Q2


def _load_prompt(*, chapter_text: str, injected_rules: list[dict], applicable_cases: list[dict]) -> str:
    template = _PROMPT_PATH.read_text(encoding="utf-8")
    return (
        template
        .replace("{chapter_text}", chapter_text)
        .replace("{injected_rules}", json.dumps(injected_rules, ensure_ascii=False, indent=2))
        .replace("{applicable_cases}", json.dumps(applicable_cases, ensure_ascii=False, indent=2))
    )


def _parse_json(text: str) -> dict[str, Any] | None:
    try:
        return json.loads(text)
    except json.JSONDecodeError:
        m = re.search(r"\{.*\}", text, re.DOTALL)
        if m:
            try:
                return json.loads(m.group(0))
            except json.JSONDecodeError:
                return None
        return None


def writer_self_check(
    *,
    chapter_text: str,
    injected_rules: list[dict],
    injected_chunks: list | None,
    applicable_cases: list[dict],
    book: str,
    chapter: str,
    llm_client: Any,
    max_retries: int = 3,
) -> ComplianceReport:
    """Compute compliance report for chapter (spec §3.3)."""
    prompt = _load_prompt(
        chapter_text=chapter_text,
        injected_rules=injected_rules,
        applicable_cases=applicable_cases,
    )

    payload: dict[str, Any] | None = None
    for _ in range(max_retries):
        try:
            resp = llm_client.messages.create(
                max_tokens=4096,
                messages=[{"role": "user", "content": prompt}],
            )
            text = resp.content[0].text
            parsed = _parse_json(text)
            if parsed and "rule_scores" in parsed:
                payload = parsed
                break
        except Exception:  # noqa: BLE001
            payload = None

    if payload is None:
        return ComplianceReport(
            rule_compliance=0.0,
            chunk_borrowing=None,
            cases_addressed=[],
            cases_violated=[c.get("case_id", "") for c in applicable_cases],
            raw_scores={"missing": [r["rule_id"] for r in injected_rules]},
            overall_passed=False,
            notes="self_check_failed",
        )

    # rule_compliance: 漏给的 rule 按 0 计
    rule_scores: dict[str, Any] = dict(payload.get("rule_scores", {}))
    missing: list[str] = []
    score_values: list[float] = []
    for rule in injected_rules:
        rid = rule["rule_id"]
        if rid in rule_scores:
            score_values.append(float(rule_scores[rid]))
        else:
            missing.append(rid)
            score_values.append(0.0)

    if score_values:
        rule_compliance = statistics.mean(score_values)
    else:
        rule_compliance = 1.0  # 无规则可违反 (Q15 注释 §3.4)
        rule_scores["__no_rules__"] = True

    # cases 二分: 漏给的 case 默认 addressed=False
    case_eval = payload.get("case_evaluation", {})
    cases_addressed: list[str] = []
    cases_violated: list[str] = []
    for case in applicable_cases:
        cid = case["case_id"]
        ev = case_eval.get(cid)
        if ev and ev.get("addressed") is True:
            cases_addressed.append(cid)
        else:
            cases_violated.append(cid)

    overall_passed = (
        rule_compliance >= _RULE_COMPLIANCE_THRESHOLD
        and len(cases_violated) == 0
    )

    raw = dict(rule_scores)
    if missing:
        raw["missing"] = missing

    return ComplianceReport(
        rule_compliance=rule_compliance,
        chunk_borrowing=None,  # M3 期 (Q15)
        cases_addressed=cases_addressed,
        cases_violated=cases_violated,
        raw_scores=raw,
        overall_passed=overall_passed,
        notes=str(payload.get("notes", "")),
    )
```

```python
"""conflict-skeleton-checker (spec §4.1)."""
from __future__ import annotations

import json
import re
from pathlib import Path
from typing import Any

from ink_writer.checkers.conflict_skeleton.models import ConflictReport

_PROMPT_PATH = Path(__file__).parent / "prompts" / "check.txt"
_BLOCK_THRESHOLD = 0.60  # spec §5.7
_MIN_CHAPTER_CHARS = 500


def _parse_json(text: str) -> dict[str, Any] | None:
    try:
        return json.loads(text)
    except json.JSONDecodeError:
        m = re.search(r"\{.*\}", text, re.DOTALL)
        if m:
            try:
                return json.loads(m.group(0))
            except json.JSONDecodeError:
                return None
        return None


def check_conflict_skeleton(
    *,
    chapter_text: str,
    book: str,
    chapter: str,
    llm_client: Any,
    max_retries: int = 3,
) -> ConflictReport:
    if len(chapter_text) < _MIN_CHAPTER_CHARS:
        return ConflictReport(
            score=0.0,
            block_threshold=_BLOCK_THRESHOLD,
            blocked=False,  # 跳过不阻断
            notes="skipped_short_chapter",
        )

    prompt = _PROMPT_PATH.read_text(encoding="utf-8").replace("{chapter_text}", chapter_text)

    payload: dict[str, Any] | None = None
    for _ in range(max_retries):
        try:
            resp = llm_client.messages.create(
                max_tokens=4096,
                messages=[{"role": "user", "content": prompt}],
            )
            text = resp.content[0].text
            parsed = _parse_json(text)
            if parsed and "has_explicit_conflict" in parsed:
                payload = parsed
                break
        except Exception:  # noqa: BLE001
            payload = None

    if payload is None:
        return ConflictReport(
            score=0.0,
            block_threshold=_BLOCK_THRESHOLD,
            blocked=True,
            notes="checker_failed",
        )

    has_conflict = bool(payload.get("has_explicit_conflict", False))
    has_three = bool(payload.get("has_three_stage_structure", False))
    count = int(payload.get("conflict_count", 0))

    score = (
        (0.5 if has_conflict else 0.0)
        + (0.3 if has_three else 0.0)
        + (min(count / 2, 1.0) * 0.2)
    )
    blocked = score < _BLOCK_THRESHOLD

    return ConflictReport(
        has_explicit_conflict=has_conflict,
```

```python
"""protagonist-agency-checker (spec §4.2)."""
from __future__ import annotations

import json
import re
from pathlib import Path
from typing import Any

from ink_writer.checkers.protagonist_agency.models import AgencyReport

_PROMPT_PATH = Path(__file__).parent / "prompts" / "check.txt"
_BLOCK_THRESHOLD = 0.60
_MIN_CHAPTER_CHARS = 500


def _parse_json(text: str) -> dict[str, Any] | None:
    try:
        return json.loads(text)
    except json.JSONDecodeError:
        m = re.search(r"\{.*\}", text, re.DOTALL)
        if m:
            try:
                return json.loads(m.group(0))
            except json.JSONDecodeError:
                return None
        return None


def check_protagonist_agency(
    *,
    chapter_text: str,
    protagonist_name: str,
    book: str,
    chapter: str,
    llm_client: Any,
    max_retries: int = 3,
) -> AgencyReport:
    if len(chapter_text) < _MIN_CHAPTER_CHARS:
        return AgencyReport(
            score=0.0,
            block_threshold=_BLOCK_THRESHOLD,
            blocked=False,
            notes="skipped_short_chapter",
        )

    prompt = (
        _PROMPT_PATH.read_text(encoding="utf-8")
        .replace("{protagonist_name}", protagonist_name)
        .replace("{chapter_text}", chapter_text)
    )

    payload: dict[str, Any] | None = None
    for _ in range(max_retries):
        try:
            resp = llm_client.messages.create(
                max_tokens=4096,
                messages=[{"role": "user", "content": prompt}],
            )
            text = resp.content[0].text
            parsed = _parse_json(text)
            if parsed and "has_active_decision" in parsed:
                payload = parsed
                break
        except Exception:  # noqa: BLE001
            payload = None

    if payload is None:
        return AgencyReport(
            score=0.0,
            block_threshold=_BLOCK_THRESHOLD,
            blocked=True,
            notes="checker_failed",
        )

    has_active = bool(payload.get("has_active_decision", False))
    has_drive = bool(payload.get("has_plot_drive", False))
    count = int(payload.get("decision_count", 0))

    score = (
        (0.5 if has_active else 0.0)
        + (0.3 if has_drive else 0.0)
        + (min(count / 2, 1.0) * 0.2)
    )
    blocked = score < _BLOCK_THRESHOLD

    return AgencyReport(
        has_active_decision=has_active,
        has_plot_drive=has_drive,
        decision_count=count,
        decision_summaries=list(payload.get("decisions", [])),
        score=score,
        block_threshold=_BLOCK_THRESHOLD,
        blocked=blocked,
        cases_hit=[],
        notes=str(payload.get("notes", "")),
    )
```

```python
"""block_threshold_wrapper — 升级现有 prompt-based checker 的输出结构 (spec §5.1).

不动现有 checker prompt 算法；wrapper 接收 score → 加 blocked / cases_hit /
would_have_blocked 字段，由 rewrite_loop orchestrator 统一消费。
"""
from __future__ import annotations

from dataclasses import dataclass, field
from typing import Any


@dataclass
class CheckerOutcome:
    checker_id: str
    score: float
    block_threshold: float
    blocked: bool
    would_have_blocked: bool = False  # dry-run 期间标记
    cases_hit: list[str] = field(default_factory=list)
    notes: str = ""


def _query_cases_by_tags(case_store: Any, tags: list[str]) -> list[str]:
    """Query case_library by tags → cases_hit."""
    if case_store is None:
        return []
    try:
        all_ids: set[str] = set()
        for tag_dict in tags:
            tag = tag_dict.get("tag") if isinstance(tag_dict, dict) else tag_dict
            if not tag:
                continue
            ids = case_store.list_ids_by_tag(tag) if hasattr(case_store, "list_ids_by_tag") else []
            all_ids.update(ids)
        return sorted(all_ids)
    except Exception:  # noqa: BLE001
        return []


def apply_block_threshold(
    *,
    checker_id: str,
    score: float,
    cfg: dict,
    is_dry_run: bool,
    case_store: Any | None,
) -> CheckerOutcome:
    """Wrap a checker score with block_threshold + cases_hit semantics."""
    checker_cfg = cfg.get(checker_id)
    if checker_cfg is None:
        return CheckerOutcome(
            checker_id=checker_id,
            score=score,
            block_threshold=0.0,
            blocked=False,
            notes=f"checker '{checker_id}' not in thresholds yaml; using default (no block)",
        )

    block_threshold = float(checker_cfg.get("block_threshold", 0.0))
    bound_cases_tags = checker_cfg.get("bound_cases", [])

    raw_blocked = score < block_threshold
    blocked = raw_blocked and not is_dry_run
    would_have_blocked = raw_blocked

    cases_hit = _query_cases_by_tags(case_store, bound_cases_tags) if raw_blocked else []

    return CheckerOutcome(
        checker_id=checker_id,
        score=score,
        block_threshold=block_threshold,
        blocked=blocked,
        would_have_blocked=would_have_blocked,
        cases_hit=cases_hit,
    )
```

```python
"""build_polish_prompt — case_id 驱动重写的 prompt 构造 (spec §5.3 + Q12)."""
from __future__ import annotations

from typing import Any


_TEMPLATE = """你是网文章节定向重写助手。给定一段章节 + 一个具体的失败模式（病例），
针对该病例重写章节，使其规避该失败模式。

输入：
- 当前章节正文：见末尾 ===CHAPTER===
- 病例 case_id: {case_id}
- 病例失败模式: {case_failure_description}
- 病例修复建议（observable 字段）: {observable_list}
- 相关 corpus chunks（M3 期为空）: {chunks_section}

要求：
1. 只针对该 case 重写最小必要段落，不要全章重写
2. 保留原章节其他部分不动
3. 输出完整重写后的章节正文（不输出 diff）
4. 末尾附 1 行说明：修改了哪几段（便于溯源）

不要包裹 markdown，直接输出重写后正文。

===CHAPTER===
{chapter_text}
===CHAPTER_END===
"""


def build_polish_prompt(
```

```python
"""rewrite_loop orchestrator — write → check → block → polish (1 case/round) → ... (spec §5.2)."""
from __future__ import annotations

from dataclasses import dataclass, field
from pathlib import Path
from typing import Any, Callable

from ink_writer.evidence_chain import EvidenceChain


_SEVERITY_ORDER = ["P0", "P1", "P2", "P3"]


@dataclass
class RewriteLoopResult:
    final_text: str
    evidence: EvidenceChain
    outcome: str  # delivered | needs_human_review
    rounds: int = 0
    history: list[str] = field(default_factory=list)  # rewrite history (r0..rN)


def collect_blocking_cases(
    compliance: Any,
    check_results: list[Any],
    case_store: Any,
) -> list[Any]:
    """Collect + dedupe + sort by severity (spec §5.4)."""
    all_ids: list[str] = []
    if not getattr(compliance, "overall_passed", True):
        all_ids.extend(getattr(compliance, "cases_violated", []))
    for r in check_results:
        if getattr(r, "blocked", False):
            all_ids.extend(getattr(r, "cases_hit", []))

    seen: set[str] = set()
    unique_cases: list[Any] = []
    for cid in all_ids:
        if cid in seen:
            continue
        seen.add(cid)
        try:
            unique_cases.append(case_store.load(cid))
        except Exception:  # noqa: BLE001
            continue

    def _severity_idx(case: Any) -> int:
        sev = getattr(getattr(case, "severity", None), "value", "P3")
        return _SEVERITY_ORDER.index(sev) if sev in _SEVERITY_ORDER else 99

    unique_cases.sort(key=_severity_idx)
    return unique_cases


def run_rewrite_loop(
    *,
    book: str,
    chapter: str,
    chapter_text: str,
    cfg: dict,
    case_store: Any,
    self_check_fn: Callable[..., Any],
    checkers_fn: Callable[..., list[Any]],
    polish_fn: Callable[..., str],
    is_dry_run: bool,
    base_dir: Path | None = None,
) -> RewriteLoopResult:
    """Main loop (spec §5.2)."""
    max_rounds = int(cfg["rewrite_loop"]["max_rounds"])
    evidence = EvidenceChain(book=book, chapter=chapter, dry_run=is_dry_run)
    current_text = chapter_text
    history = [current_text]
    polish_count = 0

    for round_idx in range(max_rounds + 1):
        compliance = self_check_fn(chapter_text=current_text, book=book, chapter=chapter)
        evidence.record_self_check(round_idx, {
            "rule_compliance": getattr(compliance, "rule_compliance", 0.0),
            "chunk_borrowing": getattr(compliance, "chunk_borrowing", None),
            "cases_addressed": list(getattr(compliance, "cases_addressed", [])),
            "cases_violated": list(getattr(compliance, "cases_violated", [])),
            "overall_passed": getattr(compliance, "overall_passed", False),
            "notes": getattr(compliance, "notes", ""),
        })

        check_results = checkers_fn(chapter_text=current_text, book=book, chapter=chapter)
        evidence.record_checkers(round_idx, [
            {
                "id": getattr(r, "checker_id", "?"),
                "score": getattr(r, "score", 0),
                "blocked": getattr(r, "blocked", False),
                "cases_hit": list(getattr(r, "cases_hit", [])),
            }
            for r in check_results
        ])

        blocking_cases = collect_blocking_cases(compliance, check_results, case_store)
        if not blocking_cases:
            evidence.outcome = "delivered"
            return RewriteLoopResult(
                final_text=current_text,
                evidence=evidence,
                outcome="delivered",
                rounds=round_idx + 1,
                history=history,
            )

        if round_idx >= max_rounds:
            evidence.outcome = "needs_human_review"
            return RewriteLoopResult(
                final_text=current_text,
                evidence=evidence,
                outcome="needs_human_review",
                rounds=round_idx + 1,
                history=history,
            )

        target = blocking_cases[0]  # severity 排序后第一个
        polish_count += 1
        current_text = polish_fn(
            chapter_text=current_text,
            case_id=target.case_id,
            case_failure_description=getattr(target.failure_pattern, "description", ""),
            case_observable=list(getattr(target.failure_pattern, "observable", [])),
            related_chunks=None,  # M3 期 chunks 为空
        )
        evidence.record_polish(round_idx + 1, target.case_id, "rewrite_for_single_case")
        history.append(current_text)

    # 不应到达
    evidence.outcome = "needs_human_review"
    return RewriteLoopResult(
        final_text=current_text,
        evidence=evidence,
        outcome="needs_human_review",
        rounds=max_rounds + 1,
        history=history,
    )
```

```python
"""dry-run mode counter + auto-switch (spec §5.6 + Q10)."""
from __future__ import annotations

from pathlib import Path


_COUNTER_FILE = ".dry_run_counter"


def _counter_path(base_dir: Path | None) -> Path:
    if base_dir is None:
        base_dir = Path("data")
    return base_dir / _COUNTER_FILE


def read_dry_run_counter(*, base_dir: Path | None = None) -> int:
    p = _counter_path(base_dir)
    if not p.is_file():
        return 0
    try:
        return int(p.read_text(encoding="utf-8").strip())
    except (ValueError, OSError):
        return 0


def increment_dry_run_counter(*, base_dir: Path | None = None) -> int:
    p = _counter_path(base_dir)
    p.parent.mkdir(parents=True, exist_ok=True)
    n = read_dry_run_counter(base_dir=base_dir) + 1
    p.write_text(str(n), encoding="utf-8")
    return n


def is_dry_run(cfg: dict, *, base_dir: Path | None = None) -> bool:
    dr = cfg.get("dry_run", {})
    if not dr.get("enabled", False):
        return False
    counter = read_dry_run_counter(base_dir=base_dir)
    if (
        counter >= dr.get("observation_chapters", 5)
        and dr.get("switch_to_block_after", True)
    ):
        return False
    return True
```

```python
"""needs_human_review.jsonl 兜底 + 4 版保留 (spec §5.5)."""
from __future__ import annotations

import json
from datetime import datetime, timezone
from pathlib import Path


def save_rewrite_history(
    *,
    book: str,
    chapter: str,
    history: list[str],
    base_dir: Path | None = None,
) -> list[Path]:
    """Save each rewrite version to data/<book>/chapters/<chapter>.r{N}.txt."""
    if base_dir is None:
        base_dir = Path(".")
    chapters_dir = base_dir / "data" / book / "chapters"
    chapters_dir.mkdir(parents=True, exist_ok=True)
    paths: list[Path] = []
    for i, text in enumerate(history):
        p = chapters_dir / f"{chapter}.r{i}.txt"
        p.write_text(text, encoding="utf-8")
        paths.append(p)
    return paths


def write_human_review_record(
    *,
    book: str,
    chapter: str,
    blocking_cases: list[str],
    rewrite_attempts: int,
    rewrite_history_paths: list[Path],
    evidence_chain_path: Path,
    base_dir: Path | None = None,
) -> Path:
    """Append needs_human_review record to data/<book>/needs_human_review.jsonl."""
    if base_dir is None:
        base_dir = Path(".")
    out_dir = base_dir / "data" / book
    out_dir.mkdir(parents=True, exist_ok=True)
    out_path = out_dir / "needs_human_review.jsonl"

    record = {
        "book": book,
        "chapter": chapter,
        "blocking_cases": list(blocking_cases),
        "rewrite_attempts": rewrite_attempts,
        "final_chapter_path": f"data/{book}/chapters/{chapter}.txt",
        "rewrite_history": [str(p) for p in rewrite_history_paths],
        "evidence_chain_path": str(evidence_chain_path),
        "marked_at": datetime.now(timezone.utc).isoformat(),
    }
    with open(out_path, "a", encoding="utf-8") as fp:
        fp.write(json.dumps(record, ensure_ascii=False))
        fp.write("\n")
    return out_path
```

```python
"""generate_dry_run_report — 5 章 dry-run 后聚合 metrics 出 markdown 报告 (spec §6.3)."""
from __future__ import annotations

import json
from collections import Counter
from datetime import datetime, timezone
from pathlib import Path
from typing import Any


def _iter_evidence(book: str, base_dir: Path) -> list[dict[str, Any]]:
    chapters_dir = base_dir / "data" / book / "chapters"
    if not chapters_dir.is_dir():
        return []
    out: list[dict[str, Any]] = []
    for p in sorted(chapters_dir.glob("*.evidence.json")):
        try:
            out.append(json.loads(p.read_text(encoding="utf-8")))
        except (json.JSONDecodeError, OSError):
            continue
    return out


def aggregate_dry_run_metrics(*, book: str, base_dir: Path) -> dict[str, Any]:
    """Aggregate metrics from all evidence_chain.json under data/<book>/chapters/."""
    chapters = _iter_evidence(book, base_dir)
    outcomes = Counter(c.get("outcome", "?") for c in chapters)

    case_hit_counter: Counter = Counter()
    rule_compliance_values: list[float] = []
    rewrite_count = 0
    for c in chapters:
        for upd in c.get("case_evidence_updates", []):
            case_hit_counter[upd.get("case_id", "")] += 1
        rounds = c.get("phase_evidence", {}).get("writer_agent", {}).get("rounds", [])
        if rounds:
            rc = rounds[0].get("compliance_report", {}).get("rule_compliance")
            if isinstance(rc, (int, float)):
                rule_compliance_values.append(float(rc))
        rewrite_count += c.get("phase_evidence", {}).get("polish_agent", {}).get("rewrite_rounds", 0)

    return {
        "total_chapters": len(chapters),
        "delivered": outcomes.get("delivered", 0),
        "needs_human_review": outcomes.get("needs_human_review", 0),
        "rewrite_rounds_total": rewrite_count,
        "rewrite_rate": (rewrite_count / len(chapters)) if chapters else 0.0,
        "human_review_rate": (outcomes.get("needs_human_review", 0) / len(chapters)) if chapters else 0.0,
        "case_hit_top10": case_hit_counter.most_common(10),
        "rule_compliance_avg": (sum(rule_compliance_values) / len(rule_compliance_values)) if rule_compliance_values else 0.0,
    }


def generate_dry_run_report(*, book: str, base_dir: Path | None = None) -> Path:
    if base_dir is None:
        base_dir = Path(".")
    metrics = aggregate_dry_run_metrics(book=book, base_dir=base_dir)
    ts = datetime.now(timezone.utc).strftime("%Y%m%dT%H%M%S")
    report_path = base_dir / "data" / f"dry_run_report_{ts}.md"
    report_path.parent.mkdir(parents=True, exist_ok=True)

    lines: list[str] = [
        "# M3 Dry-Run Report",
        f"",
        f"**Book**: {book}",
        f"**Generated at**: {datetime.now(timezone.utc).isoformat()}",
        f"**Total chapters**: {metrics['total_chapters']}",
        f"",
        "## Outcomes",
        f"- delivered: {metrics['delivered']}",
        f"- needs_human_review: {metrics['needs_human_review']}",
        f"- rewrite rounds total: {metrics['rewrite_rounds_total']}",
        f"- rewrite rate: {metrics['rewrite_rate']:.1%}",
        f"- human review rate: {metrics['human_review_rate']:.1%}",
        f"- avg rule_compliance: {metrics['rule_compliance_avg']:.3f}",
        f"",
        "## Top 10 Case Hits",
    ]
    if metrics["case_hit_top10"]:
        for cid, n in metrics["case_hit_top10"]:
            lines.append(f"- {cid}: {n}")
    else:
        lines.append("(no case hits)")

    if metrics["total_chapters"] == 0:
        lines.append("\n*0 chapters yet; run more chapters to populate this report.*")

    report_path.write_text("\n".join(lines), encoding="utf-8")
    return report_path
```
