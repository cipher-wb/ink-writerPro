- [ ] **Step 3: 写 planning_writer.py**

```python
"""planning_evidence_chain.json 写盘 + 强制必带门禁（M4 spec §6）。

ink-init / ink-plan Step 99 跑完必调 ``write_planning_evidence_chain``；
缺则 ``PlanningEvidenceChainMissingError``，让 ink-init / ink-plan 直接退出。
"""

from __future__ import annotations

import json
from pathlib import Path

from ink_writer.evidence_chain.models import EvidenceChain

DEFAULT_BASE_DIR = Path("data")


class PlanningEvidenceChainMissingError(RuntimeError):
    """ink-init / ink-plan 缺 planning_evidence_chain.json：必须立即终止。"""


def _planning_path(*, book: str, base_dir: Path | None) -> Path:
    base = Path(base_dir) if base_dir is not None else DEFAULT_BASE_DIR
    return base / book / "planning_evidence_chain.json"


def write_planning_evidence_chain(
    *,
    book: str,
    evidence: EvidenceChain,
    base_dir: Path | str | None = None,
) -> Path:
    """写到 ``<base_dir>/<book>/planning_evidence_chain.json``。

    若文件已存在（先 ink-init 后 ink-plan），合并 checkers 列表 + 用最新 stage。
    """
    if evidence.phase != "planning":
        raise ValueError(
            f"write_planning_evidence_chain expects phase='planning', got {evidence.phase!r}"
        )
    out_path = _planning_path(book=book, base_dir=Path(base_dir) if base_dir else None)
    out_path.parent.mkdir(parents=True, exist_ok=True)

    new_dict = evidence.to_dict()
    if out_path.exists():
        with open(out_path, encoding="utf-8") as fh:
            existing = json.load(fh)
        # 合并 checkers，stage 列表化
        existing.setdefault("stages", [])
        existing["stages"].append({
            "stage": new_dict.get("stage"),
            "started_at": new_dict.get("started_at"),
            "ended_at": new_dict.get("ended_at"),
            "checkers": new_dict.get("checkers", []),
            "dry_run": new_dict.get("dry_run", False),
            "skipped": new_dict.get("skipped", False),
            "skip_reason": new_dict.get("skip_reason"),
            "overall_passed": new_dict.get("overall_passed", False),
        })
        existing["phase"] = "planning"
        existing["overall_passed"] = all(s["overall_passed"] for s in existing["stages"])
        merged = existing
    else:
        merged = {
            "schema_version": "1.0",
            "phase": "planning",
            "book": book,
            "stages": [
                {
                    "stage": new_dict.get("stage"),
                    "started_at": new_dict.get("started_at"),
                    "ended_at": new_dict.get("ended_at"),
                    "checkers": new_dict.get("checkers", []),
                    "dry_run": new_dict.get("dry_run", False),
                    "skipped": new_dict.get("skipped", False),
                    "skip_reason": new_dict.get("skip_reason"),
                    "overall_passed": new_dict.get("overall_passed", False),
                }
            ],
            "overall_passed": new_dict.get("overall_passed", False),
        }

    with open(out_path, "w", encoding="utf-8") as fh:
        json.dump(merged, fh, ensure_ascii=False, indent=2)
    return out_path


def require_planning_evidence_chain(
    *,
    book: str,
    base_dir: Path | str | None = None,
) -> dict:
    """门禁：返回 dict 或 raise。"""
    out_path = _planning_path(book=book, base_dir=Path(base_dir) if base_dir else None)
    if not out_path.exists():
        raise PlanningEvidenceChainMissingError(
            f"planning_evidence_chain.json missing at {out_path}; "
            "ink-init / ink-plan Step 99 必须跑完才能继续"
        )
    with open(out_path, encoding="utf-8") as fh:
        return json.load(fh)
```

```python
"""check_genre_novelty() — M4 ink-init 题材新颖度检查（spec §3.1）。

输出 GenreNoveltyReport：
  - score = 1.0 - max(top5_similarities)
  - blocked = score < block_threshold（默认 0.40）
  - LLM/JSON 失败 max_retries 后 → score=0.0、blocked=True、notes="checker_failed"
  - cases_hit 默认空，由 planning_review 在阻断时按 case_id 注入。
"""

from __future__ import annotations

import json
from pathlib import Path
from typing import Any

from ink_writer.checkers.genre_novelty.models import GenreNoveltyReport

PROMPT_PATH = Path(__file__).resolve().parent / "prompts" / "check.txt"

_DEFAULT_MAX_TOKENS = 4096
_DEFAULT_MODEL = "glm-4.6"
_DEFAULT_BLOCK_THRESHOLD = 0.40
_TOP5 = 5


def _load_prompt() -> str:
    with open(PROMPT_PATH, encoding="utf-8") as fh:
        return fh.read()


def _build_prompt(*, genre_tags: list[str], main_plot_one_liner: str, top200: list[dict]) -> str:
    return _load_prompt().format(
        genre_tags=json.dumps(genre_tags, ensure_ascii=False),
        main_plot_one_liner=main_plot_one_liner,
        top200_json=json.dumps(top200, ensure_ascii=False),
    )


def _extract_json(raw: str) -> list[dict[str, Any]]:
    if not isinstance(raw, str):
        raise ValueError("llm response is not a string")
    raw = raw.strip()
    if raw.startswith("```"):
        raw = raw.strip("`").strip()
        if raw.lower().startswith("json"):
            raw = raw[4:].strip()
    parsed = json.loads(raw)
    if not isinstance(parsed, list):
        raise ValueError(f"expected JSON array, got {type(parsed).__name__}")
    return parsed


def check_genre_novelty(
    *,
    genre_tags: list[str],
    main_plot_one_liner: str,
    top200: list[dict],
    llm_client: Any,
    block_threshold: float = _DEFAULT_BLOCK_THRESHOLD,
    model: str = _DEFAULT_MODEL,
    max_retries: int = 2,
) -> GenreNoveltyReport:
    if not top200:
        return GenreNoveltyReport(
            score=1.0, blocked=False, top5_similar=[],
            notes="empty_top200_skipped",
        )

    prompt = _build_prompt(
        genre_tags=genre_tags,
        main_plot_one_liner=main_plot_one_liner,
        top200=top200,
    )

    last_err: Exception | None = None
    for _ in range(max_retries + 1):
        try:
            raw = llm_client.complete(
                prompt=prompt,
                model=model,
                max_tokens=_DEFAULT_MAX_TOKENS,
                temperature=0,
            )
            parsed = _extract_json(raw)
            ranked = sorted(parsed, key=lambda x: float(x.get("similarity", 0.0)), reverse=True)
            top5 = ranked[:_TOP5]
            # enrich top5 with title from top200
            title_by_rank = {item["rank"]: item.get("title", "") for item in top200}
            for entry in top5:
                entry["title"] = title_by_rank.get(entry.get("rank"), "")
            max_sim = float(top5[0]["similarity"]) if top5 else 0.0
            score = 1.0 - max_sim
            return GenreNoveltyReport(
                score=score,
                blocked=score < block_threshold,
                top5_similar=top5,
            )
        except Exception as exc:  # noqa: BLE001
            last_err = exc
    return GenreNoveltyReport(
        score=0.0, blocked=True, top5_similar=[],
        notes=f"checker_failed: {last_err}",
    )
```

"""check_naming_style() — M4 ink-init 起名风格检查（spec §3.3，纯规则无 LLM）。"""

from __future__ import annotations
import json
from pathlib import Path
from typing import Any

from ink_writer.checkers.naming_style.models import NamingStyleReport

_DEFAULT_BLOCK_THRESHOLD = 0.70
_DEFAULT_DICT_PATH = Path("data/market_intelligence/llm_naming_blacklist.json")


def _load_blacklist(path: Path) -> dict:
    with open(path, encoding="utf-8") as fh:
        return json.load(fh)


def _score_one_name(name: str, bl: dict) -> tuple[float, str]:
    """exact → 0；双字模式 → 0.4；单字模式 → 0.7；无命中 → 1.0"""
    if name in bl.get("exact_blacklist", []):
        return 0.0, "exact"
    if len(name) >= 2:
        first, second = name[0], name[1]
        first_hit = first in bl.get("char_patterns", {}).get("first_char_overused", [])
        second_hit = second in bl.get("char_patterns", {}).get("second_char_overused", [])
        if first_hit and second_hit:
            return 0.4, "double_char"
        if first_hit or second_hit:
            return 0.7, "single_char"
    return 1.0, "clean"


def check_naming_style(
    *,
    character_names: list[dict],     # [{role, name}, ...]
    blacklist_path: Path | str | None = None,
    block_threshold: float = _DEFAULT_BLOCK_THRESHOLD,
) -> NamingStyleReport:
    if not character_names:
        return NamingStyleReport(score=1.0, blocked=False, per_name_scores=[], notes="no_names")

    path = Path(blacklist_path) if blacklist_path else _DEFAULT_DICT_PATH
    if not path.exists():
        return NamingStyleReport(
            score=0.0, blocked=True, per_name_scores=[],
            notes=f"blacklist_missing: {path}",
        )
    bl = _load_blacklist(path)

    per_name = []
    for entry in character_names:
        name = entry.get("name", "")
        if not name:
            continue
        sc, hit_type = _score_one_name(name, bl)
        per_name.append({
            "role": entry.get("role", ""),
            "name": name,
            "score": sc,
            "hit_type": hit_type,
        })

    if not per_name:
        return NamingStyleReport(score=1.0, blocked=False, per_name_scores=[], notes="empty_names")

    score = sum(p["score"] for p in per_name) / len(per_name)
    return NamingStyleReport(
        score=score,
        blocked=score < block_threshold,
        per_name_scores=per_name,
    )
```

"""check_protagonist_motive() — M4 ink-init 主角动机 3 维度评估（spec §3.4）。"""

from __future__ import annotations
import json
from pathlib import Path
from typing import Any

from ink_writer.checkers.protagonist_motive.models import ProtagonistMotiveReport

PROMPT_PATH = Path(__file__).resolve().parent / "prompts" / "check.txt"
_DEFAULT_MODEL = "glm-4.6"
_DEFAULT_BLOCK_THRESHOLD = 0.65
_DIMS = ["resonance", "specific_goal", "inner_conflict"]


def _load_prompt() -> str:
    return Path(PROMPT_PATH).read_text(encoding="utf-8")


def _extract_json(raw: str) -> dict[str, Any]:
    raw = raw.strip()
    if raw.startswith("```"):
        raw = raw.strip("`").strip()
        if raw.lower().startswith("json"):
            raw = raw[4:].strip()
    return json.loads(raw)


def check_protagonist_motive(
    *, description: str, llm_client: Any,
    block_threshold: float = _DEFAULT_BLOCK_THRESHOLD,
    model: str = _DEFAULT_MODEL, max_retries: int = 2,
) -> ProtagonistMotiveReport:
    if not description or len(description) < 20:
        return ProtagonistMotiveReport(
            score=0.0, blocked=True, dim_scores={d: 0.0 for d in _DIMS},
            notes="description_too_short",
        )
    prompt = _load_prompt().format(description=description)

    last_err: Exception | None = None
    for _ in range(max_retries + 1):
        try:
            raw = llm_client.complete(prompt=prompt, model=model, max_tokens=1024, temperature=0)
            parsed = _extract_json(raw)
            dim_scores = {d: float(parsed.get(d, 0.0)) for d in _DIMS}
            score = sum(dim_scores.values()) / len(_DIMS)
            return ProtagonistMotiveReport(
                score=score,
                blocked=score < block_threshold,
                dim_scores=dim_scores,
                notes=str(parsed.get("notes", "")),
            )
        except Exception as exc:  # noqa: BLE001
            last_err = exc
    return ProtagonistMotiveReport(
        score=0.0, blocked=True, dim_scores={d: 0.0 for d in _DIMS},
        notes=f"checker_failed: {last_err}",
    )
```

"""Fetch 起点 top200 简介库 → data/market_intelligence/qidian_top200.jsonl

合规：
- 简介为公开数据
- 1 req/s 限速
- UA 礼貌（声明用途）
- 遵守 robots.txt（脚本启动前 check）

断点续爬：
- checkpoint 写入 .qidian_top200_progress（已抓 rank 列表）
- 失败重试 3 次后跳过单条记录

用法：
    python -m scripts.market_intelligence.fetch_qidian_top200 [--target 200]
"""

from __future__ import annotations

import argparse
import json
import logging
import sys
import time
import urllib.parse
import urllib.robotparser
from pathlib import Path
from typing import Any

from ink_writer.scripts.runtime_compat import enable_windows_utf8_stdio

logger = logging.getLogger(__name__)

USER_AGENT = (
    "ink-writer/M4 (educational/non-commercial; "
    "contact: insectwb@gmail.com)"
)
QIDIAN_RANK_URL = "https://www.qidian.com/rank/yuepiao/"
PROGRESS_PATH = Path("data/market_intelligence/.qidian_top200_progress")
OUT_PATH = Path("data/market_intelligence/qidian_top200.jsonl")
RATE_LIMIT_SECONDS = 1.0
MAX_RETRIES = 3


def _check_robots_txt() -> bool:
    rp = urllib.robotparser.RobotFileParser()
    rp.set_url("https://www.qidian.com/robots.txt")
    try:
        rp.read()
        return rp.can_fetch(USER_AGENT, QIDIAN_RANK_URL)
    except Exception as exc:  # noqa: BLE001
        logger.warning("robots.txt check failed (默认允许): %s", exc)
        return True


def _load_progress() -> set[int]:
    if not PROGRESS_PATH.exists():
        return set()
    return {int(line.strip()) for line in PROGRESS_PATH.read_text(encoding="utf-8").splitlines() if line.strip()}


def _save_progress_one(rank: int) -> None:
    PROGRESS_PATH.parent.mkdir(parents=True, exist_ok=True)
    with open(PROGRESS_PATH, "a", encoding="utf-8") as fh:
        fh.write(f"{rank}\n")


def _fetch_one_book(rank: int, book_id: str) -> dict[str, Any] | None:
    """Returns {rank, title, author, url, genre_tags, intro_one_liner, intro_full, fetched_at}."""
    import requests
    from bs4 import BeautifulSoup

    url = f"https://book.qidian.com/info/{book_id}"
    headers = {"User-Agent": USER_AGENT, "Accept-Language": "zh-CN"}
    for attempt in range(MAX_RETRIES):
        try:
            resp = requests.get(url, headers=headers, timeout=10)
            resp.raise_for_status()
            soup = BeautifulSoup(resp.text, "html.parser")
            title = soup.select_one(".book-info h1 em")
            author = soup.select_one(".book-info h1 a")
            intro = soup.select_one("#book-intro-detail") or soup.select_one(".book-intro p")
            tags = [t.get_text(strip=True) for t in soup.select(".book-info .tag a")]
            intro_text = intro.get_text(strip=True, separator=" ") if intro else ""
            return {
                "rank": rank,
                "title": title.get_text(strip=True) if title else "",
                "author": author.get_text(strip=True) if author else "",
                "url": url,
                "genre_tags": tags,
                "intro_one_liner": intro_text[:80],
                "intro_full": intro_text,
                "fetched_at": time.strftime("%Y-%m-%dT%H:%M:%SZ", time.gmtime()),
            }
        except Exception as exc:  # noqa: BLE001
            logger.warning("rank=%s attempt=%s failed: %s", rank, attempt, exc)
            time.sleep(RATE_LIMIT_SECONDS * (attempt + 1))
    return None


def _fetch_rank_page() -> list[tuple[int, str]]:
    """月票榜首页解析 → [(rank, book_id), ...]"""
    import requests
    from bs4 import BeautifulSoup

    headers = {"User-Agent": USER_AGENT}
    out: list[tuple[int, str]] = []
    # 起点榜单可能分页；每页 20 + 共 10 页 = 200
    for page in range(1, 11):
        url = f"{QIDIAN_RANK_URL}?page={page}"
        resp = requests.get(url, headers=headers, timeout=10)
        resp.raise_for_status()
        soup = BeautifulSoup(resp.text, "html.parser")
        items = soup.select(".rank-list .book-mid-info h2 a")
        for idx, a in enumerate(items, start=(page - 1) * 20 + 1):
            href = a.get("href", "")
            book_id = href.rstrip("/").split("/")[-1]
            if book_id:
                out.append((idx, book_id))
        time.sleep(RATE_LIMIT_SECONDS)
    return out


def main(target: int = 200) -> int:
    enable_windows_utf8_stdio()
    logging.basicConfig(level=logging.INFO, format="%(asctime)s %(levelname)s %(message)s")

    if not _check_robots_txt():
        logger.error("robots.txt 不允许抓取，终止")
        return 1

    OUT_PATH.parent.mkdir(parents=True, exist_ok=True)
    done = _load_progress()
    rank_list = _fetch_rank_page()[:target]
    logger.info("解析到 %d 条 rank entries (target=%d, 已完成=%d)", len(rank_list), target, len(done))

    fout = open(OUT_PATH, "a", encoding="utf-8")
    try:
        for rank, book_id in rank_list:
            if rank in done:
                continue
            info = _fetch_one_book(rank, book_id)
            if info is None:
                logger.error("rank=%s 跳过（重试用尽）", rank)
                continue
            fout.write(json.dumps(info, ensure_ascii=False) + "\n")
            fout.flush()
            _save_progress_one(rank)
            time.sleep(RATE_LIMIT_SECONDS)
    finally:
        fout.close()

    n = sum(1 for _ in open(OUT_PATH, encoding="utf-8"))
    logger.info("完成 — qidian_top200.jsonl 共 %d 条", n)
    return 0


if __name__ == "__main__":
    parser = argparse.ArgumentParser(description="Fetch 起点 top200 简介库")
    parser.add_argument("--target", type=int, default=200)
    args = parser.parse_args()
    sys.exit(main(target=args.target))
```

"""check_golden_finger_timing() — M4 ink-plan 金手指出场时机检查（spec §3.5）。"""

from __future__ import annotations
import json
import re
from pathlib import Path
from typing import Any

from ink_writer.checkers.golden_finger_timing.models import GoldenFingerTimingReport

PROMPT_PATH = Path(__file__).resolve().parent / "prompts" / "check.txt"
_DEFAULT_MODEL = "glm-4.6"
_DEFAULT_BLOCK_THRESHOLD = 1.0     # 硬阻断
_FIRST_N_CHAPTERS = 3


def _load_prompt() -> str:
    return Path(PROMPT_PATH).read_text(encoding="utf-8")


def _regex_check(summaries: list[dict], keywords: list[str]) -> tuple[bool, int | None]:
    pattern = "|".join(re.escape(kw) for kw in keywords if kw)
    if not pattern:
        return False, None
    rx = re.compile(pattern)
    for ch in summaries[:_FIRST_N_CHAPTERS]:
        if rx.search(ch.get("summary", "")):
            return True, int(ch.get("chapter_idx", 0))
    return False, None


def _extract_json(raw: str) -> dict[str, Any]:
    raw = raw.strip()
    if raw.startswith("```"):
        raw = raw.strip("`").strip()
        if raw.lower().startswith("json"):
            raw = raw[4:].strip()
    return json.loads(raw)


def check_golden_finger_timing(
    *,
    outline_volume_skeleton: list[dict],   # [{chapter_idx, summary}, ...]
    golden_finger_keywords: list[str],
    llm_client: Any,
    block_threshold: float = _DEFAULT_BLOCK_THRESHOLD,
    model: str = _DEFAULT_MODEL, max_retries: int = 2,
) -> GoldenFingerTimingReport:
    if len(outline_volume_skeleton) < _FIRST_N_CHAPTERS:
        return GoldenFingerTimingReport(
            score=0.0, blocked=True, regex_match=False, llm_match=None, matched_chapter=None,
            notes=f"outline_too_short: {len(outline_volume_skeleton)} < 3",
        )
    if not golden_finger_keywords:
        return GoldenFingerTimingReport(
            score=0.0, blocked=True, regex_match=False, llm_match=None, matched_chapter=None,
            notes="empty_keywords",
        )

    rx_match, rx_ch = _regex_check(outline_volume_skeleton, golden_finger_keywords)
    if rx_match:
        return GoldenFingerTimingReport(
            score=1.0, blocked=False, regex_match=True, llm_match=None, matched_chapter=rx_ch,
        )

    summaries_text = "\n".join(
        f"第{ch.get('chapter_idx')}章：{ch.get('summary', '')}"
        for ch in outline_volume_skeleton[:_FIRST_N_CHAPTERS]
    )
    prompt = _load_prompt().format(
        keywords=json.dumps(golden_finger_keywords, ensure_ascii=False),
        summaries_text=summaries_text,
    )

    last_err: Exception | None = None
    for _ in range(max_retries + 1):
        try:
            raw = llm_client.complete(prompt=prompt, model=model, max_tokens=512, temperature=0)
            parsed = _extract_json(raw)
            llm_match = bool(parsed.get("matched", False))
            return GoldenFingerTimingReport(
                score=1.0 if llm_match else 0.0,
                blocked=not llm_match,
                regex_match=False,
                llm_match=llm_match,
                matched_chapter=parsed.get("matched_chapter"),
                notes=str(parsed.get("reason", "")),
            )
        except Exception as exc:  # noqa: BLE001
            last_err = exc
    return GoldenFingerTimingReport(
        score=0.0, blocked=True, regex_match=False, llm_match=None, matched_chapter=None,
        notes=f"checker_failed: {last_err}",
    )
```

from __future__ import annotations
from pathlib import Path

DEFAULT_COUNTER_PATH = Path("data/.planning_dry_run_counter")


def get_counter(path: Path | str | None = None) -> int:
    p = Path(path) if path else DEFAULT_COUNTER_PATH
    if not p.exists():
        return 0
    try:
        return int(p.read_text(encoding="utf-8").strip())
    except ValueError:
        return 0


def increment_counter(path: Path | str | None = None) -> int:
    p = Path(path) if path else DEFAULT_COUNTER_PATH
    p.parent.mkdir(parents=True, exist_ok=True)
    cur = get_counter(p)
    new = cur + 1
    p.write_text(str(new), encoding="utf-8")
    return new


def is_dry_run_active(
    *, observation_runs: int = 5, path: Path | str | None = None,
) -> bool:
    return get_counter(path) < observation_runs

"""5 次后聚合 markdown 报告（仿 M3 dry_run_report.py）。"""

from __future__ import annotations
import json
from collections import Counter
from pathlib import Path


def generate_planning_dry_run_report(
    *, base_dir: Path | str = "data",
) -> str:
    base = Path(base_dir)
    rows = []
    case_counter: Counter[str] = Counter()
    checker_scores: dict[str, list[float]] = {}
    for book_dir in base.iterdir():
        if not book_dir.is_dir():
            continue
        ev_path = book_dir / "planning_evidence_chain.json"
        if not ev_path.exists():
            continue
        ev = json.loads(ev_path.read_text(encoding="utf-8"))
        for stage in ev.get("stages", []):
            for ck in stage.get("checkers", []):
                checker_scores.setdefault(ck["checker"], []).append(ck.get("score", 0.0))
                for cid in ck.get("cases_hit", []) or []:
                    case_counter[cid] += 1
            rows.append({
                "book": book_dir.name,
                "stage": stage.get("stage"),
                "passed": stage.get("overall_passed"),
                "skipped": stage.get("skipped"),
            })

    md = ["# M4 Planning Dry-Run Report\n"]
    md.append(f"**books reviewed**: {len({r['book'] for r in rows})}\n")
    md.append("## 各 checker 平均分\n")
    for c, scores in sorted(checker_scores.items()):
        md.append(f"- {c}: mean={sum(scores)/len(scores):.3f} (n={len(scores)})")
    md.append("\n## case 触发频次（上游 cases）\n")
    if case_counter:
        for cid, n in case_counter.most_common():
            md.append(f"- {cid}: {n} 次")
    else:
        md.append("- (none)")
    md.append("\n## per-stage rows\n")
    for r in rows:
        md.append(f"- book={r['book']} stage={r['stage']} passed={r['passed']} skipped={r['skipped']}")
    return "\n".join(md) + "\n"


if __name__ == "__main__":
    print(generate_planning_dry_run_report())
